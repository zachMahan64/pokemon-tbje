import java.util.List;

public class Pokemon {
    private String name;
    private String type;
    private int baseAttack;
    private int baseDefense;
    private int baseHp;
    private List<Move> moves;  // List of moves available to this Pokemon

    // Constructor
    public Pokemon(Species species) {
        this.name = species.getName();
        this.type = species.getType();
        this.baseAttack = species.getBaseAttack();
        this.baseDefense = species.getBaseDefense();
        this.baseHp = species.getBaseHp();
        this.moves = species.getMoves();  // Set the moves from the species
    }

    // Getter for moves
    public List<Move> getMoves() {
        return this.moves;
    }

    // Method to get a move by name (for user input)
    public Move getMove(String moveName) {
        for (Move move : moves) {
            if (move.getName().equalsIgnoreCase(moveName)) {
                return move;  // Return the move if names match
            }
        }
        return null;  // Return null if no matching move found
    }

    // Getters for other attributes
    public String getName() {
        return name;
    }

    public String getType() {
        return type;
    }

    public int getAttack() {
        return baseAttack;
    }

    public int getDefense() {
        return baseDefense;
    }

    public int getHp() {
        return baseHp;
    }

    @Override
    public String toString() {
        return name + " | Type: " + type + " | HP: " + baseHp + " | Attack: " + baseAttack + " | Defense: " + baseDefense;
    }
}
