import java.awt.desktop.ScreenSleepEvent;
import java.util.*;
public class Game {
static boolean playStatus = true;
static String username;
    public static void main(String[] args) {
        Scanner sc1 = new Scanner(System.in);
        triggerOpening(sc1);
        enterPlayMenu(sc1);
        while (playStatus = true){
            enterMainMenu(sc1);
        }
        sc1.close();
    }

    //cutscenes
    public static void triggerOpening(Scanner sc1){
        try {
            playOpening(sc1);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
    public static void playOpening(Scanner sc1) throws InterruptedException {
        Graphics.printBigTitleImage();
        int openingWaitTime = 2000; //0 for testing, 2000 when finished
        System.out.println("                         TEXT-BASED JAVA EDITION\n");
        Thread.sleep(openingWaitTime);
        Graphics.printProfessorOak();
        System.out.println("Welcome to the world of Pokemon! I'm Professor Oak.");
        Thread.sleep(openingWaitTime);
        System.out.println("I have invented a marvelous new technology: the Pokedex!");
        Thread.sleep(openingWaitTime);
        System.out.println("It lets us track and identify every known Pokemon! Isn't that amazing?");
        Thread.sleep(openingWaitTime);
        System.out.println("And, since you're embarking on your own Pokemon journey, I want to give one to you.");
        Thread.sleep(openingWaitTime);
        System.out.println("\nObtained the Pokedex from Professor Oak!\n");
        Thread.sleep(openingWaitTime);
        System.out.println("Oh, and, uh, remind me, what's your name?");
        String username = getUsername(sc1);
        System.out.println("Right, of course, " + username + "! What starter pokemon would you like? \n [1] Bulbasaur [4] Charmander [7] Squirtle");
        int starterNum = getStarter(sc1);
        Pokemon starter = null;
        switch (starterNum) {
            case 1:
                starter = new Pokemon(Species.getSpecies("Bulbasaur"));
                Graphics.printBulbsaur();
                break;
            case 4:
                starter = new Pokemon(Species.getSpecies("Charmander"));
                Graphics.printCharmander();
                break;
            case 7:
                starter = new Pokemon(Species.getSpecies("Squirtle"));
                Graphics.printSquirtle();
                break;
            default:
                System.out.println("Invalid choice! You must choose [1], [4], or [7].");
                return;
        }
        Thread.sleep(openingWaitTime);
        System.out.println("Ah, " + starter.getName() + "! Such exquisite taste.");
        Party.addToParty(starter);
    }

    //menu logic
    public static void enterMainMenu(Scanner sc1) {
        Graphics.printBigTitleImage();
        System.out.println("                         TEXT-BASED JAVA EDITION\n");

        while (true) {
            System.out.println("Type \"S\" to start, \"P\" to see the Pokedex, or \"E\" to exit the game.");
            String choice = sc1.nextLine().trim().toUpperCase();

            if (choice.equals("S")) {
                System.out.println("Starting the game...\n");
                enterPlayMenu(sc1);
                break;
            } else if (choice.equals("P")) {
                Pokedex.printPokedex();
            } else if (choice.equals("E")) {
                System.out.println("Exiting game...");
                System.exit(0);
            } else {
                System.out.println("Invalid input. Please try again.");
            }
        }
    }
    public static void enterPlayMenu(Scanner sc1) {
        System.out.println("Queueing Pokemon Sound...\n");

        while (true) {
            System.out.println("Type \"R\" to go onto a route, \"V\" to view your party, \"P\" to see the Pokedex, \"M\" to return to the main menu, or \"E\" to exit the game.");
            String choice = sc1.nextLine().trim().toUpperCase();
            if (choice.equals("R")) {
                while(true){
                    goOnRoute(sc1);
                }
            } else if (choice.equals("V")) {
                Party.printParty();;
            } else if (choice.equals("P")) {
                Pokedex.printPokedex();
            } else if (choice.equals("M")) {
                enterMainMenu(sc1);
                break;
            } else if (choice.equals("E")) {
                System.out.println("Exiting game...");
                System.exit(0);
            } else {
                System.out.println("Invalid input. Please try again.");
            }
        }
    }

    //location logic
    public static void goOnRoute(Scanner sc1) {
        System.out.println("Do you want to walk into the tall grass in front of you? Press \"W\" to take a few steps, or press \"T\" to turn back.");

        while (true) {
            String choiceGrass = sc1.nextLine().trim().toUpperCase();

            if (choiceGrass.equals("W")) {
                Encounter.enterWildPkmMenu(sc1);
                break;
            } else if (choiceGrass.equals("T")) {
                System.out.println("You decided not to enter the tall grass.");
                enterPlayMenu(sc1);
                break;
            } else {
                System.out.println("Invalid input. Please type W or T.");
            }
        }
    }

    //info (get______)
    public static String getUsername(Scanner sc1){
        String username = sc1.nextLine();
        return username;
    }
    public static int getStarter(Scanner sc1) {
        int starter;
        while (true) {
            if (sc1.hasNextInt()) {
                starter = sc1.nextInt();
                sc1.nextLine();  // Consume the leftover newline character
                if (starter == 1 || starter == 4 || starter == 7) {
                    return starter;
                }
            } else {
                sc1.next();  // Skip non-integer input
            }
            System.out.println("Invalid choice. Please enter 1, 4, or 7.");
        }
    }
}
