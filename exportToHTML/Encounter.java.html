import java.util.List;
import java.util.Random;
import java.util.Scanner;

public class Encounter {
    public static Pokemon genWildPkm() {
        Species wildSpecies = null;
        switch (getEncNum()) {
            case 1: wildSpecies = Species.getSpecies("Bulbasaur"); break;
            case 2: wildSpecies = Species.getSpecies("Ivysaur"); break;
            case 3: wildSpecies = Species.getSpecies("Venusaur"); break;
            case 4: wildSpecies = Species.getSpecies("Charmander"); break;
            case 5: wildSpecies = Species.getSpecies("Charmeleon"); break;
            case 6: wildSpecies = Species.getSpecies("Charizard"); break;
            case 7: wildSpecies = Species.getSpecies("Squirtle"); break;
            case 8: wildSpecies = Species.getSpecies("Wartortle"); break;
            case 9: wildSpecies = Species.getSpecies("Blastoise"); break;
            case 10: wildSpecies = Species.getSpecies("Caterpie"); break;
            case 11: wildSpecies = Species.getSpecies("Metapod"); break;
            case 12: wildSpecies = Species.getSpecies("Butterfree"); break;
            case 13: wildSpecies = Species.getSpecies("Weedle"); break;
            case 14: wildSpecies = Species.getSpecies("Kakuna"); break;
            case 15: wildSpecies = Species.getSpecies("Beedrill"); break;
            case 16: wildSpecies = Species.getSpecies("Pidgey"); break;
            case 17: wildSpecies = Species.getSpecies("Pidgeotto"); break;
            case 18: wildSpecies = Species.getSpecies("Pidgeot"); break;
            case 19: wildSpecies = Species.getSpecies("Rattata"); break;
            case 20: wildSpecies = Species.getSpecies("Raticate"); break;
            case 21: wildSpecies = Species.getSpecies("Spearow"); break;
            case 22: wildSpecies = Species.getSpecies("Fearow"); break;
            case 23: wildSpecies = Species.getSpecies("Ekans"); break;
            case 24: wildSpecies = Species.getSpecies("Arbok"); break;
            case 25: wildSpecies = Species.getSpecies("Pikachu"); break;
            case 26: wildSpecies = Species.getSpecies("Raichu"); break;
            default:
                System.out.println("Unknown Pokémon encountered!");
                wildSpecies = null;
                break;
        } //gets wildSpecies from encNum
        Pokemon wildPokemon = new Pokemon(wildSpecies);
        System.out.println("A wild " + wildPokemon.getName() + " appeared!");
        return wildPokemon;
    }

    public static void enterWildPkmMenu(Scanner sc1) {
        Pokemon wildPokemon = genWildPkm();
        playWildPkm(wildPokemon, sc1);
    }
    public static void playWildPkm(Pokemon wildPokemon, Scanner sc1) {
        while (true) {
            System.out.println("\nFight [F] | Bag [B] | Switch [S] | Run [R]");
            String encMenuChoice = sc1.nextLine().trim().toUpperCase();
            if (encMenuChoice.equals("F")) {
                System.out.println("[F] What will " + Party.pkm1.getName() + " do?");
                printMoves(Party.pkm1);
                String moveChoice = sc1.nextLine().trim().toUpperCase();
                Move moveToUse = Party.pkm1.getMove(moveChoice);
                if (moveToUse != null) {
                    System.out.println(Party.pkm1.getName() + " used " + moveToUse.getName() + "!");
                } else {
                    System.out.println("Invalid move!");
                }
            } else if (encMenuChoice.equals("B")) {
                System.out.println("You opened your Bag!");
            } else if (encMenuChoice.equals("S")) {
                Party.switchPokemon(sc1);
            } else if (encMenuChoice.equals("R")) {
                System.out.println("You ran away from the wild " + wildPokemon.getName() + ".");
                break;
            } else {
                System.out.println("Invalid input.");
            }
        }
    }
    public static int getEncNum() {
        Random rand = new Random();
        return rand.nextInt(26) + 1;
    }
    // Helper function to list all moves of a Pokémon
    private static void printMoves(Pokemon pokemon) {
        System.out.println("Choose a move:");
        List<Move> moves = pokemon.getMoves();
        for (Move move : moves) {
            System.out.println("[" + move.getName() + "] Damage: " + move.getDamage());
        }
    }
}
