    public static String getName(int pokedexNum){
        if (pokedexNum < 1 || pokedexNum > 26) {
            return "Invalid Pokedex number";
        }
        else {
            String[] pokedex = {
                    "Bulbasaur", "Ivysaur", "Venusaur", "Charmander", "Charmeleon", "Charizard",
