import java.util.Scanner;

public class Party {
    public static Pokemon pkm1 = null;
    public static Pokemon pkm2 = null;
    public static Pokemon pkm3 = null;
    public static Pokemon pkm4 = null;
    public static Pokemon pkm5 = null;
    public static Pokemon pkm6 = null;

    public static void addToParty(Pokemon pokemon) {
        if (pkm1 == null) {
            pkm1 = pokemon;
        } else if (pkm2 == null) {
            pkm2 = pokemon;
        } else if (pkm3 == null) {
            pkm3 = pokemon;
        } else if (pkm4 == null) {
            pkm4 = pokemon;
        } else if (pkm5 == null) {
            pkm5 = pokemon;
        } else if (pkm6 == null) {
            pkm6 = pokemon;
        } else {
            System.out.println("Your party is full! Cannot add " + pokemon.getName() + ".");
        }
    }

    public static void printParty() {
        System.out.println("Your Pok√©mon Party:");
        if (pkm1 != null) {
            System.out.println("1. " + pkm1.getName() + " | HP: " + pkm1.getHp() + " | Attack: " + pkm1.getAttack() + " | Defense: " + pkm1.getDefense());
        }
        if (pkm2 != null) {
            System.out.println("2. " + pkm2.getName() + " | HP: " + pkm2.getHp() + " | Attack: " + pkm2.getAttack() + " | Defense: " + pkm2.getDefense());
        }
        if (pkm3 != null) {
            System.out.println("3. " + pkm3.getName() + " | HP: " + pkm3.getHp() + " | Attack: " + pkm3.getAttack() + " | Defense: " + pkm3.getDefense());
        }
        if (pkm4 != null) {
            System.out.println("4. " + pkm4.getName() + " | HP: " + pkm4.getHp() + " | Attack: " + pkm4.getAttack() + " | Defense: " + pkm4.getDefense());
        }
        if (pkm5 != null) {
            System.out.println("5. " + pkm5.getName() + " | HP: " + pkm5.getHp() + " | Attack: " + pkm5.getAttack() + " | Defense: " + pkm5.getDefense());
        }
        if (pkm6 != null) {
            System.out.println("6. " + pkm6.getName() + " | HP: " + pkm6.getHp() + " | Attack: " + pkm6.getAttack() + " | Defense: " + pkm6.getDefense());
        }

        if (pkm1 == null && pkm2 == null && pkm3 == null && pkm4 == null && pkm5 == null && pkm6 == null) {
            System.out.println("Your party is empty.");
        }
    }

    public static void switchPokemon(Scanner sc1) {
        int slot;
        Pokemon selectedPokemon = null;

        // Loop until a valid selection is made
        while (true) {
            slot = askToSwitchPokemon(sc1);

            // If user chose to go back, exit method
            if (slot == -1) {
                System.out.println("Switching canceled.");
                return;
            }

            // Assign selected Pok√©mon
            switch (slot) {
                case 2: selectedPokemon = pkm2; break;
                case 3: selectedPokemon = pkm3; break;
                case 4: selectedPokemon = pkm4; break;
                case 5: selectedPokemon = pkm5; break;
                case 6: selectedPokemon = pkm6; break;
                default:
                    System.out.println("Invalid choice. Please choose again.");
                    continue;
            }

            if (selectedPokemon != null) {
                break; // Valid Pok√©mon found, exit loop
            } else {
                System.out.println("No Pok√©mon in that slot! Choose a different slot.");
            }
        }

        // Swap the active Pok√©mon (pkm1) with the selected one
        switch (slot) {
            case 2: pkm2 = pkm1; break;
            case 3: pkm3 = pkm1; break;
            case 4: pkm4 = pkm1; break;
            case 5: pkm5 = pkm1; break;
            case 6: pkm6 = pkm1; break;
        }
        pkm1 = selectedPokemon;

        System.out.println("Switched to " + pkm1.getName() + "!");
    }

    public static int askToSwitchPokemon(Scanner sc1) {
        while (true) {
            printParty();
            System.out.println("Enter a Pok√©mon (2-6) to switch to, or 'C' to cancel:");

            String input = sc1.nextLine().trim(); // Use nextLine() to consume newline
            if (input.equalsIgnoreCase("C")) {
                return -1; // Return -1 to indicate cancellation
            }
            try {
                int choiceSwitchPokemon = Integer.parseInt(input);
                if (choiceSwitchPokemon >= 2 && choiceSwitchPokemon <= 6) {
                    return choiceSwitchPokemon; // Return valid choice
                }
            } catch (NumberFormatException e) {
            }
            System.out.println("Invalid input! Please enter a number between 2-6 or 'C' to cancel.");
        }
    }
}
