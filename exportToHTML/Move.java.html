public class Move {
    private String name;
    private String type;
    private int damage;

    // Constructor to initialize the move's name, type, and damage
    public Move(String name, String type, int damage) {
        this.name = name;
        this.type = type;
        this.damage = damage;
    }

    // Getter for the name of the move
    public String getName() {
        return name;
    }

    // Getter for the type of the move (e.g., "Normal", "Fire", etc.)
    public String getType() {
        return type;
    }

    // Getter for the damage of the move
    public int getDamage() {
        return damage;
    }

    // Optional: String representation of the move
    @Override
    public String toString() {
        return name + " (" + type + " Type, Damage: " + damage + ")";
    }
}
