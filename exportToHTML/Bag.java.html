public class Bag {
    private int potions;
    private int revives;
    private int pokeballs;

    public Bag() {
        potions = 0;
        revives = 0;
        pokeballs = 0;
    }

    public void addPotion(int amount) {
        if (amount > 0) {
            potions += amount;
            System.out.println(amount + " Potion(s) added to your bag.");
        } else {
            System.out.println("Invalid amount for Potions.");
        }
    }

    public void addRevive(int amount) {
        if (amount > 0) {
            revives += amount;
            System.out.println(amount + " Revive(s) added to your bag.");
        } else {
            System.out.println("Invalid amount for Revives.");
        }
    }

    public void addPokeball(int amount) {
        if (amount > 0) {
            pokeballs += amount;
            System.out.println(amount + " Pokéball(s) added to your bag.");
        } else {
            System.out.println("Invalid amount for Pokéballs.");
        }
    }

    public void usePotion() {
        if (potions > 0) {
            potions--;
            System.out.println("You used a Potion.");
        } else {
            System.out.println("You have no Potions left!");
        }
    }

    public void useRevive() {
        if (revives > 0) {
            revives--;
            System.out.println("You used a Revive.");
        } else {
            System.out.println("You have no Revives left!");
        }
    }

    public void usePokeball() {
        if (pokeballs > 0) {
            pokeballs--;
            System.out.println("You threw a Pokéball.");
        } else {
            System.out.println("You have no Pokéballs left!");
        }
    }

    public void checkBag() {
        System.out.println("Your Bag contains:");
        System.out.println("Potions: " + potions);
        System.out.println("Revives: " + revives);
        System.out.println("Pokéballs: " + pokeballs);
    }

    // Getters for each item
    public int getPotions() {
        return potions;
    }
    public int getRevives() {
        return revives;
    }
    public int getPokeballs() {
        return pokeballs;
    }
}
